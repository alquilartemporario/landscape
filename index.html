"left" : "right"; } canvas.addEventListener("touchstart",(e)=>{ unlockAudio(); leftPressed = false; rightPressed = false; for(let t of e.touches){ if(side(t.clientX)==="left") leftPressed = true; if(side(t.clientX)==="right") rightPressed = true; } }); canvas.addEventListener("touchmove",(e)=>{ leftPressed = false; rightPressed = false; for(let t of e.touches){ if(side(t.clientX)==="left") leftPressed = true; if(side(t.clientX)==="right") rightPressed = true; } }); canvas.addEventListener("touchend",()=>{ leftPressed = false; rightPressed = false; });
